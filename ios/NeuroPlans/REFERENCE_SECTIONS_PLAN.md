```swift
// New models

struct NeurologyExam: Codable, Identifiable {
    let id: String
    let title: String
    let category: String           // "core", "specialized", "rapid"
    let steps: [ExamStep]
    let normalFindings: String
    let abnormalFindings: [String]
    let clinicalPearls: [String]
    let relatedPlans: [String]     // Links back to clinical plans
}

struct ExamStep: Codable, Identifiable {
    let id: String
    let instruction: String
    let technique: String?
    let normalResult: String?
    let abnormalResults: [AbnormalFinding]?
    let image: String?             // Asset name for illustration
}

struct ClinicalScale: Codable, Identifiable {
    let id: String
    let title: String
    let abbreviation: String
    let category: String           // "stroke", "cognitive", "neuromuscular", etc.
    let description: String
    let components: [ScaleComponent]
    let interpretation: [ScoreRange]
    let citation: String?
    let relatedPlans: [String]
}

struct ScaleComponent: Codable, Identifiable {
    let id: String
    let name: String
    let options: [ScoreOption]
}

struct ScoreOption: Codable {
    let label: String
    let value: Int
    let description: String?
}

struct ScoreRange: Codable {
    let min: Int
    let max: Int
    let label: String              // "Minor stroke", "Moderate", "Severe"
    let color: String              // For visual indicator
    let action: String?            // Clinical recommendation
}

struct NeurologyTool: Codable, Identifiable {
    let id: String
    let title: String
    let category: String           // "calculator", "anatomy", "lab", "protocol", "conversion"
    let inputs: [ToolInput]
    let formula: String?           // For calculators
    let content: ToolContent       // For references/protocols
    let relatedPlans: [String]
}
```
